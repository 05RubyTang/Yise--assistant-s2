### 2. 主应用：`src/App.jsx`

#### 核心功能

**路由管理**（基于栈的导航）

```jsx
function AppInner() {
  const [pageStack, setPageStack] = useState([{ name: 'home', params: {} }])
  const current = pageStack[pageStack.length - 1]

  // Tab页面直接跳转（清空栈）
  // 其他页面压栈（可返回）
  const navigate = (name, params) => {
    if (TAB_PAGES.includes(name)) {
      setPageStack([{ name, params }])
    } else {
      setPageStack(prev => [...prev, { name, params }])
    }
  }

  const goBack = () => {
    setPageStack(prev => prev.length > 1 ? prev.slice(0, -1) : prev)
  }
}
```

**页面渲染**（Switch 路由）

```jsx
switch (current.name) {
  case 'home':        return <Home navigate={navigate} />
  case 'plans':       return <PlanList navigate={navigate} mode="library" />
  case 'checklist':   return <Checklist planId={current.params.planId} ... />
  case 'recorder':    return <Recorder planId={current.params.planId} ... />
  case 'collection':  return <Collection />
  case 'profile':     return <Profile navigate={navigate} />
  // ... 其他 10+ 页面
}
```

**布局结构**


#### 核心 Actions

```jsx
dispatch({ type: 'MARK_SHINY_OBTAINED', shinyName, from, season })
dispatch({ type: 'START_TASK', plan, season })
dispatch({ type: 'RECORD_BREAK', taskId, result, spiritName, pool })
dispatch({ type: 'COMPLETE_TASK', taskId, result, resultSpirit, season })

dispatch({ type: 'SWITCH_SEASON', season })
```

#### 持久化策略

**1. localStorage（本地优先）**
```jsx
// 每次 state 变化自动保存
useEffect(() => {
  localStorage.setItem(STORAGE_KEY, JSON.stringify(state))
}, [state])
```

**2. Supabase（云端同步）**
```jsx
// 防抖同步（5秒延迟）
const doSyncNow = useCallback(debounce(() => {
  supabase.from('user_data').upsert({
    user_id: state.supabaseUserId,
    data: state,
    updated_at: new Date()
  })
}, 5000), [state])
```

**3. 脏标记机制（防丢失）**
```jsx
// 上传前标记
localStorage.setItem(DIRTY_KEY, 'true')

// 上传成功清除
localStorage.removeItem(DIRTY_KEY)

// 启动时检查补传
if (localStorage.getItem(DIRTY_KEY) === 'true') {
  doSyncNow()
}
```
